108 def net_electric_field(self, position: Point2D) -> float:
109 """
110 Calculate the net electric field magnitude at a point
111
112 Args:
113 position (Point2D): The position to measure the electric field at
114
115 Returns:
116 float: magnitude of electric field
117 """
118
119 return np.sqrt(self.electric_field_x(position)**2 + self.electric_field_y(position)**2)

121 def electric_field_x(self, position: Point2D) -> float:
122 """
123 Calculate the x component of the electric field at a point.
124
125 Args:
126 position (Point2D): The position to measure the electric field at.
127
128 Returns:
129 float: x component of electric field.
130 """
131
132 e_x = 0.0
133
134 # Sum up electric field x component for each charge
135 for charge in self.charges:
136 x_inc = charge.electric_field_x(position)
137 e_x += x_inc if np.isfinite(x_inc) else 0.0
138
139 return e_x
