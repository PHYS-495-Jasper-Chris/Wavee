26 def __init__(self, x_coef: float, y_coef: float, offset: float, charge_density: float) -> None:
27 """
28 Initialize the infinite line charge with its equation and charge density.
29
30 The line equation is written in the form ``ax + by + c = 0``, where a is ``x_coef``, b is
31 ``y_coef``, and c is ``offset``.
32
33 Args:
34 x_coef (float): The coefficient on x in the equation of the line (a).
35 y_coef (float): The coefficient on y in the equation of the line (b).
36 offset (float): The offset in the equation of the line (c).
37 charge_density (float): The charge density, in C/m.
38
39 Raises:
40 RuntimeError: Cannot have both the x and y coefficient equating to zero.
41 """
42
43 if x_coef == 0 and y_coef == 0:
44 raise RuntimeError(
45 "Invalid infinite line charge equation: x_coef and y_coef cannot simultaneously be "
46 "0")
47
48 self.x_coef = x_coef
49 self.y_coef = y_coef
50 self.offset = offset
51 self.charge_density = charge_density

53 def electric_field_magnitude(self, point: Point2D) -> float:
54 """
55 The net magnitude of the electric field at the given point.
56
57 Args:
58 point (Point2D): The point to measure the field at.
59
60 Returns:
61 float: The net (signed) magnitude of the electric field at the given point.
62 """
63
64 radial_distance = self._radial_distance(point)
65
66 # Make sure we don't divide by 0
67 if radial_distance == 0.0:
68 return 0.0
69
70 # E = 2k λ/r
71 return 2 * COULOMB_CONSTANT * self.charge_density / radial_distance

73 def electric_field_x(self, point: Point2D) -> float:
74 """
75 The x component of the electric field at a given point.
76
77 Args:
78 point (Point2D): The point to measure the field at.
79
80 Returns:
81 float: Magnitude of the electric field's x-component due to the infinite line charge at
82 this point.
83 """
84
85 # The direction of the electric field is completely orthogonal to the direction of the line
86 # charge itself, so add a pi/2 rotation to the angle.
87 magnitude = self.electric_field_magnitudeelectric_field_magnitude(point) * np.cos(self._line_angle() + np.pi / 2)
88
89 if self._flip_direction(point):
90 magnitude *= -1
91
92 return magnitude

156 def electric_field_mag_eqn(self) -> sympy.Basic:
157 """
158 Returns the position-independent electric field equation for this infinite line charge.
159
160 Returns:
161 Basic: sympy representation of the signed magnitude of the electric field.
162 """
163
164 # E = 2k λ/r
165 # radial distance
166 r_sym = (abs(self.x_coef * x + self.y_coef * y + self.offset)
167 / sympy.sqrt(self.x_coef**2 + self.y_coef**2))
168 return 2 * COULOMB_CONSTANT_SYM * self.charge_density / r_sym

206 def _radial_distance(self, point: Point2D) -> float:
207 """
208 The shortest distance from a point to the infinite line of charge.
209
210 Args:
211 point (Point2D): The point to measure the distance from.
212
213 Returns:
214 float: minimal radial distance from ``point`` to line charge.
215 """
216
217 return (abs(self.x_coef * point.x + self.y_coef * point.y + self.offset)
218 / np.sqrt(self.x_coef**2 + self.y_coef**2))

220 def _closest_point(self, point: Point2D) -> Point2D:
221 """
222 The closest point on the line from a given point.
223
224 Args:
225 point (Point2D): The point to find the closest point to.
226
227 Returns:
228 Point2D: The closest point.
229 """
230
231 x_pos = (self.y_coef * (self.y_coef * point.x - self.x_coef * point.y)
232 - self.x_coef * self.offset) / (self.x_coef**2 + self.y_coef**2)
233
234 y_pos = (self.x_coef * (self.x_coef * point.y - self.y_coef * point.x)
235 - self.y_coef * self.offset) / (self.x_coef**2 + self.y_coef**2)
236
237 return Point2D(x_pos, y_pos)

239 def _line_angle(self) -> float:
240 """
241 Get the angle of the line as though it was through the origin, in radians.
242
243 Returns:
244 float: The angle of the infinite line charge itself, in radians.
245 """
246
247 if self.y_coef == 0.0:
248 # Only have x component, so let the line point straight up (independent of x's sign).
249 return 3 * np.pi / 2
250
251 if self.x_coef == 0.0:
252 # Only have y component, so let the line point straight in the +x direction (independent
253 # of y's sign).
254 return 0
255
256 # The slope of the line, in ay + bx + c = 0 turns in to y = -b/a x - c/a
257 return np.arctan(-self.x_coef / self.y_coef)

259 def _flip_direction(self, point: Point2D) -> bool:
260 """
261 Returns whether to flip the direction of a component of a magnitude, based on the location
262 of the point.
263
264 Returns:
265 bool: True if the magnitude value should be negated, False otherwise.
266 """
267
268 # Now we need to flip the direction if we are on the opposite side of the line. This means
269 # that there are 4 cases, part of 2 groups: the x component and y component are both greater
270 # than the closest point on the line; the x and y component are both less than the closest
271 # point on the line; and one (but not both) of the components is greater than the closest
272 # point on the line.
273
274 closest_point = self._closest_point(point)
275
276 if closest_point.x >= point.x and closest_point.y >= point.y:
277 return True
278
279 if closest_point.x <= point.x and closest_point.y <= point.y:
280 return False
281
282 if closest_point.x >= point.x and closest_point.y <= point.y:
283 return False
284
285 return True

287 def _closest_point_eqn(self) -> Tuple[sympy.Basic, sympy.Basic]:
288 """
289 Return the formula for the closest point to a general x, y position.
290
291 Returns:
292 Tuple[Basic, Basic]: The x position and y position as sympy objects.
293 """
294
295 x_pos = (self.y_coef * (self.y_coef * x - self.x_coef * y)
296 - self.x_coef * self.offset) / (self.x_coef**2 + self.y_coef**2)
297
298 y_pos = (self.x_coef * (self.x_coef * y - self.y_coef * x)
299 - self.y_coef * self.offset) / (self.x_coef**2 + self.y_coef**2)
300
301 return x_pos, y_pos

303 def _flip_direction_eqn(self) -> Tuple[sympy.Basic, sympy.Basic]:
304 """
305 The inequalities for the positive and negative equations.
306
307 Returns:
308 Tuple[Basic, Basic]: The positive and negative inequalities to be used, as boolean
309 compositions of relationals.
310 """
311
312 x_closest, y_closest = self._closest_point_eqn()
313
314 if self.x_coef == 0:
315 pos_eq = clean_inequality(y_closest <= y, y)
316 neg_eq = clean_inequality(y_closest >= y, y)
317 elif self.y_coef == 0:
318 pos_eq = clean_inequality(x_closest <= x, x)
319 neg_eq = clean_inequality(x_closest >= x, x)
320 else:
321 pos_eq = sympy.Or(clean_inequality([x_closest <= x, y_closest <= y], x),
322 clean_inequality([x_closest >= x, y_closest <= y], x)).simplify()
323 neg_eq = sympy.Or(clean_inequality([x_closest >= x, y_closest >= y], x),
324 clean_inequality([x_closest <= x, y_closest >= y], x)).simplify()
325
326 return pos_eq, neg_eq
