204 def _mouse_moved(self, event: Tuple[QtCore.QPointF]) -> None:
205 """
206 Callback executed when the mouse moves in the graph widget.
207
208 Args:
209 event (Tuple[QPointF]): A tuple containing the position of the mouse.
210 """
211
212 pos = event[0]
213 if self.graph_widget.sceneBoundingRect().contains(pos):
214 mouse_point: QtCore.QPointF = self.graph_widget.get_pi_vb()[1].mapSceneToView(pos)
215 x_pos, y_pos = mouse_point.x(), mouse_point.y()
216 ef_mag_net = self.graph_widget.graph_window.net_electric_field(Point2D(x_pos, y_pos))
217
218 x_fs = "e" if abs(x_pos) > 1e5 or (x_pos != 0.0 and abs(x_pos) < 1e-2) else "f"
219 y_fs = "e" if abs(y_pos) > 1e5 or (y_pos != 0.0 and abs(y_pos) < 1e-2) else "f"
220 text = f"X: {x_pos:+.2{x_fs}} Y: {y_pos:+.2{y_fs}} Magnitude: {ef_mag_net:.6g}"
221
222 self.graph_widget.setToolTip(text)
223 self.status_bar.showMessage(text)

236 def _update_equations(self) -> None:
237 """
238 Read the values from the equations thread and update the labels.
239 """
240
241 mag_eqns = self.equations_thread.mag_eqns
242 x_eqns = self.equations_thread.x_eqns
243 y_eqns = self.equations_thread.y_eqns
244
245 mag_html = ""
246 for i, mag_eqn in enumerate(mag_eqns):
247 mag_html += f"E_{i}={latex(mag_eqn)},"
248
249 x_html = "E_x(x,y)="
250 for i, x_eqn in enumerate(x_eqns):
251 if len(x_eqns) > 1:
252 x_html += f"\\left({latex(x_eqn)}\\right)+"
253 else:
254 x_html += f"{latex(x_eqn)}+"
255
256 y_html = "E_y(x,y)="
257 for i, y_eqn in enumerate(y_eqns):
258 if len(y_eqns) > 1:
259 y_html += f"\\left({latex(y_eqn)}\\right)+"
260 else:
261 y_html += f"{latex(y_eqn)}+"
262
263 self.net_mag_equation_label.setHtml(make_source(mag_html[:-1]) if mag_eqns else "")
264 self.x_equation_label.setHtml(make_source(x_html[:-1]) if x_eqns else "")
265 self.y_equation_label.setHtml(make_source(y_html[:-1]) if y_eqns else "")
